# ORBIT_ANIMATION
Uses manim library to display animations explaining orbit determination.

please install manim using pip before installing requirements.

manimgl -lq -q earth_satellite.py -c black

manimgl -w -l -q --video_dir low_quality earth_satellite.py -c black

manimgl -w -m -q --video_dir medium_quality earth_satellite.py -c black

manimgl -w --hd -q --video_dir high_quality earth_satellite.py -c black
